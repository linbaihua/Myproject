
# 缓存路由
          
<!-- 保持子组件活跃不被销毁，否则组件会自动被销毁 -->
<!-- 单个include就是包含全部 -->
<keep-alive include>
    <router-view></router-view>
</keep-alive>

<!-- 只包含一个组件 -->
<keep-alive include='xxxx'>
    <router-view></router-view>
</keep-alive>

<!-- 包含多个组件 -->
<keep-alive :include="['xxx','xxx']">
    <router-view></router-view>
</keep-alive>
